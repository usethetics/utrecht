import { Meta, Story, Canvas, ArgsTable } from "@storybook/addon-docs";
import clsx from "clsx";
import "./bem.scss";

export const defaultArgs = {
  distanced: false,
  innerHTML: "",
};

export const Template = ({ distanced = false, innerHTML = "", type }) =>
  `<div class="${clsx("utrecht-alert", {
    "utrecht-alert--distanced": distanced,
    [`utrecht-alert--${type}`]: type,
  })}">
  <div class="utrecht-alert__icon"></div>
  <div class="utrecht-alert__content">${innerHTML}</div>
</div>`;

<Meta
  title="CSS Component/Alert"
  argTypes={{
    distanced: {
      description: "Distance the element from adjacent content",
      control: "boolean",
    },
    innerHTML: {
      description: "Content of the block",
      control: "text",
    },
    status: {
      control: { type: "select" },
      description: "Status",
      options: ["neutral", "danger", "warning", "safe"],
    },
  }}
  parameters={{
    docs: {
      transformSource: (_src, { args }) => Template(args),
    },
    status: {
      type: "WORK IN PROGRESS",
    },
  }}
/>
