export const FormDataList = ({ items = [] }) =>
  `<dl class="utrecht-form-data-list">${items
    .map(
      ({ label, value }) => `
  <!-- <div class="utrecht-form-data-list__item"> -->
    <dt class="utrecht-form-data-list__label">${label}</dt>
    <dd class="utrecht-form-data-list__data">${value}</dd>
  <!-- </div> -->
  `
    )
    .join("")}
</dl>`;

<Meta
  title="CSS Component/Form Data List"
  argTypes={{
    items: {
      description: "Items in the list",
      type: {
        name: "array",
        required: true,
      },
    },
  }}
  parameters={{
    docs: {
      transformSource: (_src, { args }) => FormDataList(args),
    },
    status: {
      type: "WORK IN PROGRESS",
    },
  }}
/>
